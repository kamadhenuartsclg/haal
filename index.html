class drop
{

void disp2();

{
settextstyle(1,HORTY_DIR,5);
outtextxy(150,150,"BCA 3 YEAR");

{
for(i=1;i<=300;i++)
{
outtextxy(150,20,"BCA 3 YEAR");

}

}
}

#include stdio.h
#include conio.h
#include math.h
#include stdlib.h
#include process.h
#include graphics.h

void main()
{
clrscr();
int gd=DETECT,gm,i,j,x1,y1,x2,y2;
float dx,dy,step,xin,yin,x,y;
char q[10];
printf("\n Enter the x,y coordinate values:\n");
scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
initgraph(&gd,&gm," ");
cleardevice;
x1=x1+50;
x2=x2+50;
y1=450-y1;
y2=450-y2;
line(50,10,50,435);
outtextxy(10,15,"YAXIS");
line(50,435,450,435);
outtextxy(490,435,"XAXIS");

for(j=30;j<390;i+=30,j+=30)
{
sprintf(q,"%2d",j);
outtextxy(50+j,450,q);
}

for(j=30;j<430;i+=30,j+=30)
{
sprintf(q,"%2d",j);
outtextxy(20,450-j,q);
}

dx=x2-x1;
dy=y2-y1;
if(abs(dx)>abs(dy))
step=abs(dx);
else
step=abs(dy);

xin=dx/step;
yin=dy/step;
x=x1;
y=y1;

for(i=0;i< step;i++)
{
putpixel(x,y,WHITE);
x+=xin;
y+=yin;
}

getch();
closegraph();
}
